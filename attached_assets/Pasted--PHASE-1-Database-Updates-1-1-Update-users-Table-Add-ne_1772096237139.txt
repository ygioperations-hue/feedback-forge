✅ PHASE 2 — Role-Based Middleware

Create two middleware functions:

2.1 Require Authentication
function requireAuth(req, res, next) {
  if (!req.user) return res.status(401).json({ error: 'Unauthorized' });
  next();
}
2.2 Require Platform Admin
function requirePlatformAdmin(req, res, next) {
  if (!req.user || req.user.role !== 'platform_admin') {
    return res.status(403).json({ error: 'Forbidden' });
  }
  next();
}
2.3 Update Paywall Middleware
function requireActivePlan(req, res, next) {
  if (req.user.role === 'platform_admin') return next();

  if (req.user.plan_type === 'none') {
    return res.status(403).json({ error: 'Plan required' });
  }

  next();
}
